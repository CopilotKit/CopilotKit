---
title: "useRenderToolCall"
description: "Deprecated alias for useToolRenderingResolver"
---

<Callout type="warn">
  **Deprecated.** Use
  [`useToolRenderingResolver`](/reference/hooks/useToolRenderingResolver)
  instead. `useRenderToolCall` continues to work during the deprecation window,
  but it logs a development warning.
</Callout>

## Replacement

```tsx
import { useToolRenderingResolver } from "@copilotkit/react-core/v2";

const resolveToolRendering = useToolRenderingResolver();
```

`useToolRenderingResolver` is the same low-level resolver with a clearer name: it resolves an existing renderer for a tool call. It does not register a renderer and it does not execute a tool handler.

## Before / After

**Before:**

```tsx
import React from "react";
import type { AssistantMessage, Message, ToolMessage } from "@ag-ui/core";
import { useRenderToolCall } from "@copilotkit/react-core/v2";

function ToolCalls({
  message,
  messages,
}: {
  message: AssistantMessage;
  messages: Message[];
}) {
  const renderToolCall = useRenderToolCall();

  return (
    <>
      {message.toolCalls?.map((toolCall) => {
        const toolMessage = messages.find(
          (candidate) =>
            candidate.role === "tool" && candidate.toolCallId === toolCall.id,
        ) as ToolMessage | undefined;

        const renderedToolElement = renderToolCall({ toolCall, toolMessage });

        return renderedToolElement ? (
          <React.Fragment key={toolCall.id}>
            {renderedToolElement}
          </React.Fragment>
        ) : null;
      })}
    </>
  );
}
```

**After:**

```tsx
import React from "react";
import type { AssistantMessage, Message, ToolMessage } from "@ag-ui/core";
import { useToolRenderingResolver } from "@copilotkit/react-core/v2";

function ToolCalls({
  message,
  messages,
}: {
  message: AssistantMessage;
  messages: Message[];
}) {
  const resolveToolRendering = useToolRenderingResolver();

  return (
    <>
      {message.toolCalls?.map((toolCall) => {
        const toolMessage = messages.find(
          (candidate) =>
            candidate.role === "tool" && candidate.toolCallId === toolCall.id,
        ) as ToolMessage | undefined;

        const renderedToolElement = resolveToolRendering({
          toolCall,
          toolMessage,
        });

        return renderedToolElement ? (
          <React.Fragment key={toolCall.id}>
            {renderedToolElement}
          </React.Fragment>
        ) : null;
      })}
    </>
  );
}
```
