Before you begin, you'll need the following:

- An Anthropic API key
- Node.js 20+
- Your favorite package manager

            <Step>
                ### Expose your agent via AG-UI

                Create an agent file that exposes your Claude agent via the AG-UI protocol:

                <Tabs groupId="language" items={['TypeScript', 'Python']}>
                    <Tab value="TypeScript">
                        ```typescript title="server.ts"
                        import http from "node:http";
                        import { ClaudeAgentAdapter } from "@ag-ui/claude-agent-sdk";
                        import type { RunAgentInput } from "@ag-ui/core";
                        import { EventEncoder } from "@ag-ui/encoder";

                        const adapter = new ClaudeAgentAdapter({
                          agentId: "my_agent",
                          model: "claude-haiku-4-5",
                          systemPrompt: "You are a helpful assistant.",
                        });

                        const server = http.createServer(async (req, res) => {
                          res.setHeader("Access-Control-Allow-Origin", "*");
                          res.setHeader("Access-Control-Allow-Methods", "GET, POST, OPTIONS");
                          res.setHeader("Access-Control-Allow-Headers", "*");

                          if (req.method === "OPTIONS") {
                            res.writeHead(204);
                            res.end();
                            return;
                          }

                          if (req.method !== "POST") {
                            res.writeHead(404);
                            res.end();
                            return;
                          }

                          const chunks: Buffer[] = [];
                          for await (const chunk of req) chunks.push(chunk as Buffer);
                          const body = Buffer.concat(chunks).toString("utf-8");

                          let input: RunAgentInput;
                          try {
                            input = JSON.parse(body) as RunAgentInput;
                          } catch {
                            res.writeHead(400, { "Content-Type": "application/json" });
                            res.end(JSON.stringify({ error: "Invalid JSON body" }));
                            return;
                          }

                          const encoder = new EventEncoder({
                            accept: req.headers.accept ?? "text/event-stream",
                          });

                          res.writeHead(200, {
                            "Content-Type": encoder.getContentType(),
                            "Cache-Control": "no-cache",
                            Connection: "keep-alive",
                          });

                          adapter.run(input).subscribe({
                            next: (event) => res.write(encoder.encode(event)),
                            error: (err) => {
                              console.error(err);
                              res.end();
                            },
                            complete: () => res.end(),
                          });
                        });

                        server.listen(8000, () => console.log("Agent running on http://localhost:8000"));
                        ```
                    </Tab>
                    <Tab value="Python">
                        ```python title="server.py"
                        from fastapi import FastAPI
                        from ag_ui_claude_sdk import ClaudeAgentAdapter, add_claude_fastapi_endpoint

                        adapter = ClaudeAgentAdapter(
                            name="my_agent",
                            options={
                                "model": "claude-haiku-4-5",
                                "system_prompt": "You are a helpful assistant.",
                            },
                        )

                        app = FastAPI()
                        add_claude_fastapi_endpoint(app, adapter, path="/")

                        if __name__ == "__main__":
                            import uvicorn
                            uvicorn.run(app, host="localhost", port=8000)
                        ```
                    </Tab>
                </Tabs>

            </Step>
            <Step>
                ### Setup Copilot Runtime

                Create an API route to connect CopilotKit to your Claude Agent SDK agent:

                ```tsx title="app/api/copilotkit/route.ts"
                import {
                  CopilotRuntime,
                  ExperimentalEmptyAdapter,
                  copilotRuntimeNextJSAppRouterEndpoint,
                } from "@copilotkit/runtime";
                import { HttpAgent } from "@ag-ui/client";
                import { NextRequest } from "next/server";

                const serviceAdapter = new ExperimentalEmptyAdapter();

                const runtime = new CopilotRuntime({
                  agents: {
                    my_agent: new HttpAgent({ url: "http://localhost:8000/" }),
                  }
                });

                export const POST = async (req: NextRequest) => {
                  const { handleRequest } = copilotRuntimeNextJSAppRouterEndpoint({
                    runtime,
                    serviceAdapter,
                    endpoint: "/api/copilotkit",
                  });

                  return handleRequest(req);
                };
               ```
            </Step>
            <Step>
                ### Configure CopilotKit Provider

                Wrap your application with the CopilotKit provider:

                ```tsx title="app/layout.tsx"
                import { CopilotKit } from "@copilotkit/react-core"; // [!code highlight]
                import "@copilotkit/react-ui/v2/styles.css";

                // ...

                export default function RootLayout({ children }: {children: React.ReactNode}) {
                  return (
                    <html lang="en">
                      <body>
                        {/* [!code highlight:3] */}
                        <CopilotKit runtimeUrl="/api/copilotkit" agent="my_agent">
                          {children}
                        </CopilotKit>
                      </body>
                    </html>
                  );
                }
                ```
            </Step>
            <Step>
              ### Add the chat interface

              Add the CopilotSidebar component to your page:

              ```tsx title="app/page.tsx"
              import { CopilotSidebar } from "@copilotkit/react-core/v2"; // [!code highlight:1]

              export default function Page() {
                return (
                  <main>
                    <h1>Your App</h1>
                    {/* [!code highlight:1] */}
                    <CopilotSidebar />
                  </main>
                );
              }
              ```
            </Step>
